<script type="text/babel">
  const root = document.getElementById('root')
  function List({data, children}) {
    return (
      <ul> 
        {data.map((...props) => children(...props))}
      </ul>
    )
  }
  function App() {
    const cars = ['BMW', 'Mazda', "Toyota"]
    return (
      <React.Fragment>
        <List data={cars}>
          {(item, index) => <li key={index}>{item}</li>}
        </List>

      </React.Fragment>
    )
  } 
  
    ReactDOM.render(<App />, root)
