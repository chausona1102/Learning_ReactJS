#   React Element Types

function Header() {  
    return <div class="header">New Header</div>
  }

  class Content extends React.components {
    render() {
        return <div class="header">New Header</div>
    }
  }
  const app =
    <div class="wrapper">
      <Header/>
      <div class="content">Content</div>
      <div class="footer">Footer</div>
    </div>

  ReactDOM.render(app, document.getElementById('root'))
